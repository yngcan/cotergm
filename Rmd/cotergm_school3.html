d2ud <- function(net){
    net <- as.matrix(net)
    for (i in 1:nrow(net)){
        for(j in 1:nrow(net)){
            net[i,j] = max(net[i,j],net[j,i])
        }}
    as.network(net,directed=FALSE)
}
